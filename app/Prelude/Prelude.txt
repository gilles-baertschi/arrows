class Arrow y where
    id :: y a a
    arr :: Id a b -> y a b
    (>>>) :: (y a b, y b c) -> y a c
    first :: y a b -> y (a, c) (b, c)
    second :: y a b -> y (c, a) (c, b)
    (***) :: (y a c, y b d) -> y (a, b) (c, d)
    (&&&) :: (y a b, y a c) -> y a (b, c)

class Arrow y => Choice y where
    left :: y a b -> y (a | c) (b | c)
    right :: y a b -> y (c | a) (c | b)
    (+++) :: (y a c, y b d) -> y (a | b) (c | d)
    (|||) :: (y a c, y b c) -> y (a | b) c

class Arrow y => Apply y where
    app :: y (y a b, a) b
